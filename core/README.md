## âš¡ Quick Start

```python
from core.brain import LiveBrain
from core.state_machine import SaleStateMachine
from core.observability import MetricsCollector

# Initialize
brain = LiveBrain(config)
state_machine = SaleStateMachine()
metrics = MetricsCollector()

# Decision flow
decision = brain.decide(
    intent="price_question",
    viewer_count=150,
    time_since_last_speak=5.2,
    sale_state=state_machine.current_state
)

# Log metrics
metrics.log_decision(decision)

if decision.action == "SPEAK":
    # Generate response based on state
    response = generate_response(decision.reason, state_machine.current_state)
    
    # Transition state if needed
    state_machine.transition(decision.reason)
    
    # Track
    metrics.log_speak(response, viewer_count=150)
```
